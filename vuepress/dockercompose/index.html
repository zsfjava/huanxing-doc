# 一、概述

本文档主要是对智能设备平台在进行应用部署的一些规范及注意事项。

# 二、部署说明

# 1. 服务配置要求

硬件 要求
CPU 8 核+
内存 16G+
硬盘 60G+

# 2. 软件配置要求

软件 版本
jdk 1.8
mysql 8.0.30
redis 6.2.6
node 14
maven 3.5.4+

智能设备平台后台管理模块 device-platform-admin 5
智能设备平台后台分析模块 device-platform-analysis 6

# 三、docker-componse 部署服务

# 1. 初始化服务器

wget https://pigx.vip/os7init.sh

sh os7init.sh jycloud

# 2. 官网源

 sudo yum-config-manager \
     --add-repo \
     https://download.docker.com/linux/centos/docker-ce.repo

#更新源
 sudo yum makecache fast

# 3. 安装

 sudo yum install docker-ce

# 4. 使用加速器

curl -sSL https://get.daocloud.io/daotools/set_mirror.sh | sh -s http://2be16b36.m.daocloud.io

# 5. 安装 docker-compose

curl -L https://github.com/docker/compose/releases/download/1.25.0/docker-compose-`uname -s`-`uname -m` > /usr/local/bin/docker-compose

chmod +x /usr/local/bin/docker-compose

# 6. 启动 docker

 # 启动Docker服务
 systemctl start docker
 # 设置开机自动启动
 systemctl enable docker
 # 查看docker的版本,测试是否安装成功
 docker -v

# 7. 部署服务端

 # 项目打包
 mvn clean install -Dmaven.test.skip=true
 # 把项目打zip上传到服务器
 scp device.zip
 # 解压zip 并进入项目目录
 cd device
 # 启动容器  静静等待所有服务启动
 docker-compose up -d

# 四、前端部署

# 官网

官网：https://nginx.org/

# 1. 创建文件

vim /etc/yum.repos.d/nginx.repo

# 2. 粘贴到新创建的文件

[nginx-stable]
name=nginx stable repo
baseurl=http://nginx.org/packages/centos/$releasever/$basearch/
gpgcheck=1
enabled=1
gpgkey=https://nginx.org/keys/nginx_signing.key
module_hotfixes=true

[nginx-mainline]
name=nginx mainline repo
baseurl=http://nginx.org/packages/mainline/centos/$releasever/$basearch/
gpgcheck=1
enabled=0
gpgkey=https://nginx.org/keys/nginx_signing.key
module_hotfixes=true

# 3. 安装 nginx 安装中 全部输入 y

vim /etc/yum.repos.d/nginx.repo

# 4. 新增 device-ui.conf

# 进入 cd /etc/nginx/conf.d
vim device-ui.conf
# 粘贴
server {
    listen 80;
    server_name localhost;

    gzip on;
    gzip_static on;     # 需要http_gzip_static_module 模块
    gzip_min_length 1k;
    gzip_comp_level 4;
    gzip_proxied any;
    gzip_types text/plain text/xml text/css;
    gzip_vary on;
    gzip_disable "MSIE [1-6]\.(?!.*SV1)";

    # 前端打包好的dist目录文件
    root /root/dist/;

    location ~* ^/(code|auth|admin|monitor|platform|anlys) {
       proxy_pass http://127.0.0.1:9999;
       proxy_connect_timeout 15s;
       proxy_send_timeout 15s;
       proxy_read_timeout 15s;
       proxy_set_header X-Real-IP $remote_addr;
       proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
       proxy_set_header X-Forwarded-Proto http;
    }
    location / {
        try_files $uri $uri/ /index.html;
    }
    # 避免端点安全问题
    if ($request_uri ~ "/actuator"){
       return 403;
    }
}
# 删除默认conf
rm default.conf

# 5. 修改 nginx.conf

#修改权限  user nginx 改成 user root
vim /etc/nginx/nginx.conf
# 启动
nginx

# 6. 前端项目打包

# 安装前端依赖
npm install --registry=https://registry.npm.taobao.org
# 打包
npm run build
# 上传
scp dist.zip
# 解压
unzip dist
