    # 1. 官网：https://nginx.org/

    # 2. 创建文件 vim /etc/yum.repos.d/nginx.repo

    [nginx-stable]
name=nginx stable repo
baseurl=http://nginx.org/packages/centos/$releasever/$basearch/
gpgcheck=1
enabled=1
gpgkey=https://nginx.org/keys/nginx_signing.key
module_hotfixes=true

[nginx-mainline]
name=nginx mainline repo
baseurl=http://nginx.org/packages/mainline/centos/$releasever/$basearch/
gpgcheck=1
enabled=0
gpgkey=https://nginx.org/keys/nginx_signing.key
module_hotfixes=true

    # 3. 安装 nginx

    sudo yum install yum-utils
sudo yum-config-manager --enable nginx-mainline
sudo yum install nginx

    # 4. 进入 cd /etc/nginx/conf.d 新增 huanxing-ui.conf

    server {
        listen 80;
        # listen 443 ssl;
        server_name localhost;

        # ssl on;
        # ssl_certificate /root/ssl/ihuanxing/ihuanxing.com.pem;
        # ssl_certificate_key /root/ssl/ihuanxing/ihuanxing.com.key;

        # ssl_prefer_server_ciphers on;
        # ssl_session_timeout 60m;

        gzip  on;
        gzip_min_length  1k;
        gzip_buffers     4 16k;
        gzip_http_version 1.1;
        gzip_comp_level 9;
        gzip_types text/plain text/xml text/css;
        gzip_disable "MSIE [1-6]\.";
        gzip_vary on;

        # 打包好的dist目录文件
        root /root/dist;

        location ^~/api/ {
           proxy_pass http://huanxing-gateway:9999/; 
           proxy_connect_timeout 60s;
           proxy_send_timeout 300s;
           proxy_read_timeout 300s;
           proxy_set_header X-Real-IP $remote_addr;
           proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
           # For WebSocket upgrade header
           proxy_http_version 1.1;
           proxy_set_header Upgrade $http_upgrade;
           proxy_set_header Connection "upgrade";

        }
          # 避免端点安全问题
        if ($request_uri ~ "/actuator"){
	          return 403;
        }
    }

    # 5. 修改 vim /etc/nginx/nginx.conf

    user nginx 改为 root 权限

    # 6. 启动 nginx 服务

    nginx

    # 7. huanxing-ui 打包

    # 安装前端依赖
npm install  --registry=https://registry.npmmirror.com

# 打包 把dist包上传到指定目录文件
npm run build
