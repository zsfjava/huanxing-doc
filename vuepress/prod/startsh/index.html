#!/bin/sh
export GATEWAY=huanxing-gateway.jar
export AUTH=huanxing-auth.jar
export UPMS=huanxing-upms-admin.jar
export MALL_API=huanxing-mall-api.jar
export MALL_ADMIN=huanxing-mall-admin.jar
export MINIAPP_WEIXIN=huanxing-miniapp-weixin.jar
export PAY=huanxing-pay-api.jar

export GATEWAY_LOG=../logs/huanxing-gateway.log
export AUTH_LOG=../logs/huanxing-auth.log
export UPMS_LOG=../logs/huanxing-upms-admin.log
export MALL_API_LOG=../logs/huanxing-mall-api.log
export MALL_ADMIN_LOG=../logs/huanxing-mall-admin.log
export MINIAPP_WEIXIN_LOG=../logs/huanxing-miniapp-weixin.log
export PAY_LOG=../logs/huanxing-pay-api.log

export GATEWAY_PORT=9999
export AUTH_PORT=5227
export UPMS_PORT=5327
export MALL_API_PORT=6227
export MALL_ADMIN_PORT=6127
export MINIAPP_WEIXIN_PORT=8121
export PAY_PORT=7121

case "$1" in

start)
        ## 启动gateway
        echo "--------开始启动GATEWAY---------------"
        nohup java -Xms256m -Xmx512m -jar $GATEWAY > $GATEWAY_LOG 2>&1 &
        GATEWAY_PID=`lsof -i:$GATEWAY_PORT|grep "LISTEN"|awk '{print $2}'`
        until [ -n "$GATEWAY_PID" ]
            do
              GATEWAY_PID=`lsof -i:$GATEWAY_PORT|grep "LISTEN"|awk '{print $2}'`
            done
        echo "GATEWAY pid is $GATEWAY_PID"
        echo "---------GATEWAY 启动成功-----------"

        ## 启动auth
        echo "--------开始启动AUTH---------------"
        nohup java -Xms256m -Xmx512m -jar $AUTH > $AUTH_LOG 2>&1 &
        AUTH_PID=`lsof -i:$AUTH_PORT|grep "LISTEN"|awk '{print $2}'`
        until [ -n "$AUTH_PID" ]
            do
              AUTH_PID=`lsof -i:$AUTH_PORT|grep "LISTEN"|awk '{print $2}'`
            done
        echo "AUTH pid is $AUTH_PID"
        echo "---------AUTH 启动成功-----------"

        ## 启动UPMS
        echo "--------开始启动UPMS---------------"
        nohup java -Xms256m -Xmx512m -jar $UPMS > $UPMS_LOG 2>&1 &
        UPMS_PID=`lsof -i:$UPMS_PORT|grep "LISTEN"|awk '{print $2}'`
        until [ -n "$UPMS_PID" ]
            do
              UPMS_PID=`lsof -i:$UPMS_PORT|grep "LISTEN"|awk '{print $2}'`
            done
        echo "UPMS pid is $UPMS_PID"
        echo "---------UPMS 启动成功-----------"

        ## 启动MALL-API
        echo "--------开始启动MALL-API---------------"
        nohup java -Xms256m -Xmx512m -jar $MALL_API > $MALL_API_LOG 2>&1 &
        MALL_API_PID=`lsof -i:$MALL_API_PORT|grep "LISTEN"|awk '{print $2}'`
        until [ -n "$MALL_API_PID" ]
            do
              MALL_API_PID=`lsof -i:$MALL_API_PORT|grep "LISTEN"|awk '{print $2}'`
            done
        echo "MALL_API pid is $MALL_API_PID"
        echo "---------MALL-API 启动成功-----------"

        ## 启动MALL-ADMIN
        echo "--------开始启动MALL-ADMIN---------------"
        nohup java -Xms256m -Xmx512m -jar $MALL_ADMIN > $MALL_ADMIN_LOG 2>&1 &
        MALL_ADMIN_PID=`lsof -i:$MALL_ADMIN_PORT|grep "LISTEN"|awk '{print $2}'`
        until [ -n "$MALL_ADMIN_PID" ]
            do
              MALL_ADMIN_PID=`lsof -i:$MALL_ADMIN_PORT|grep "LISTEN"|awk '{print $2}'`
            done
        echo "MALL_ADMIN pid is $MALL_ADMIN_PID"
        echo "---------MALL-ADMIN 启动成功-----------"

        ## 启动MINIAPP-WEIXIN
        echo "--------开始启动MINIAPP-WEIXIN---------------"
        nohup java -Xms256m -Xmx512m -jar $MINIAPP_WEIXIN > $MINIAPP_WEIXIN_LOG 2>&1 &
        MINIAPP_WEIXIN_PID=`lsof -i:$MINIAPP_WEIXIN_PORT|grep "LISTEN"|awk '{print $2}'`
        until [ -n "$MINIAPP_WEIXIN_PID" ]
            do
              MINIAPP_WEIXIN_PID=`lsof -i:$MINIAPP_WEIXIN_PORT|grep "LISTEN"|awk '{print $2}'`
            done
        echo "MINIAPP_WEIXIN pid is $MINIAPP_WEIXIN_PID"
        echo "---------MINIAPP-WEIXIN 启动成功-----------"

        ## 启动PAY
        echo "--------开始启动PAY---------------"
        nohup java -Xms256m -Xmx512m -jar $PAY > $PAY_LOG 2>&1 &
        PAY_PID=`lsof -i:$PAY_PORT|grep "LISTEN"|awk '{print $2}'`
        until [ -n "$PAY_PID" ]
            do
              PAY_PID=`lsof -i:$PAY_PORT|grep "LISTEN"|awk '{print $2}'`
            done
        echo "PAY pid is $PAY_PID"
        echo "---------PAY 启动成功-----------"

        echo "===startAll success==="
        ;;

 stop)
         P_ID=`ps -ef | grep -w $UPMS | grep -v "grep" | awk '{print $2}'`
        if [ "$P_ID" == "" ]; then
            echo "===UPMS process not exists or stop success"
        else
            kill -9 $P_ID
            echo "UPMS killed success"
        fi

         P_ID=`ps -ef | grep -w $GATEWAY | grep -v "grep" | awk '{print $2}'`
        if [ "$P_ID" == "" ]; then
            echo "===GATEWAY process not exists or stop success"
        else
            kill -9 $P_ID
            echo "GATEWAY killed success"
        fi

         P_ID=`ps -ef | grep -w $AUTH | grep -v "grep" | awk '{print $2}'`
        if [ "$P_ID" == "" ]; then
            echo "===AUTH process not exists or stop success"
        else
            kill -9 $P_ID
            echo "AUTH killed success"
        fi

         P_ID=`ps -ef | grep -w $MALL_API | grep -v "grep" | awk '{print $2}'`
        if [ "$P_ID" == "" ]; then
            echo "===MALL_API process not exists or stop success"
        else
            kill -9 $P_ID
            echo "MALL_API killed success"
        fi

         P_ID=`ps -ef | grep -w $MALL_ADMIN | grep -v "grep" | awk '{print $2}'`
        if [ "$P_ID" == "" ]; then
            echo "===MALL_ADMIN process not exists or stop success"
        else
            kill -9 $P_ID
            echo "MALL_ADMIN killed success"
        fi

         P_ID=`ps -ef | grep -w $PAY | grep -v "grep" | awk '{print $2}'`
        if [ "$P_ID" == "" ]; then
            echo "===PAY process not exists or stop success"
        else
            kill -9 $P_ID
            echo "PAY killed success"
        fi

         P_ID=`ps -ef | grep -w $MINIAPP_WEIXIN | grep -v "grep" | awk '{print $2}'`
        if [ "$P_ID" == "" ]; then
            echo "===MINIAPP_WEIXIN process not exists or stop success"
        else
            kill -9 $P_ID
            echo "MINIAPP_WEIXIN killed success"
        fi

        echo "===stop success==="
        ;;

restart)
        $0 stop
        sleep 10
        $0 start
        echo "===restart success==="
        ;;
esac
exit 0
